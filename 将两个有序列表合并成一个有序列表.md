
## 递归法 ##

	def sort1(a,b):
	    c = []
	    if len(a) == 0 :
	        c.extend(b)
	        return c
	    elif len(b) == 0:
	        c.extend(a)
	        return c
	    else:
	        if a[0] < b[0]:
	            c.append(a[0])
	            del a[0]
	        else:
	            c.append(b[0])
	            del b[0]
	        c += sort1(a, b)
